\section{Chebotarev Example}
\label{code:ChebotarevExample}
\begin{minted}{sage}
m = 0
f = x^3 + m*x^2 + (m-3)*x - 1
K = NumberField(f, 'a')

print("Defining polynomial:")
print(f)
> x^3 - 3*x - 1
print("Discriminant:")
print(K.discriminant())
> 81
print("Extension Degree:")
print(K.degree())
> 3
print("Galois extension:")
print(K.is_galois())
> True
print("Basis:")
True
print(K.maximal_order().basis())
> [1, a, a^2]
\end{minted}
